import express from 'express'; import path from 'path'; export async function createServer() { const app = express(); // Serve static files from the "public" directory app.use(express.static(path.join(__dirname, '../public'))); // Fallback to index.html for SPA routing app.get('*', (req, res) => { res.sendFile(path.join(__dirname, '../public/index.html')); }); return app; }
